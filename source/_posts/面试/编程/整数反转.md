function _reverse(number) {
    // 补全代码
    if (number >= 0) {
        return number.toString().split("").reverse().join("") - 0
    } else {
        let str = number.toString()
        let flag = str[0]
        let subStr = str.slice(1).split("").reverse().join("")
        return flag + subStr - 0
    }
}
