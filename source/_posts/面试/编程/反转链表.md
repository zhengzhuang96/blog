function ReverseList(pHead)
{
    if (!pHead || !pHead.next) {
        return pHead
    }
    var pre = null
    var cur = pHead
    var next
    while (cur) {
        next = cur.next
        cur.next = pre
        pre = cur
        cur = next
    }
    return pre
}
module.exports = {
    ReverseList : ReverseList
};
